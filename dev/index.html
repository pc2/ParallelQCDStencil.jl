ParallelQCDStencil.jl

Installation

] add https://git.uni-paderborn.de/cs-hit/theses/2022_parab/qcd-stencil/ParallelQCDStencil.jl

Note: The minimal required Julia version is 1.7.

Assumptions

  • Link matrices must be of the same type and size 3x3 (independent of dimension)
  • For parallel computation:
    • Inner grids must be square and of the same size.
    • Division of the global grid by inner grid length must be an integer.

Usage

julia> using ParallelQCDStencil

Input parameters

We are using struct for passing around the parameters to the fuctions. It consists of :

L = 5                           #number of lattices in per dimension
N = L^D                         #number of sites
NL = N * D                      #number of links
LD = 3                          #link matrix length
C1::T = 2                       #stencil constant 
C2::T = 3                       #stencil constant   
IG = 2                          #inner grid length
n_IG_d = Int(floor(L / IG))     #grids per dimension
n_IG = n_IG_d^D                 #number of inner grids
master_rank = 0                 #mpi master rank

Struct instantiation :

# 2D Sequential 
p = StencilPara{Int64, 2}(L=4)

# 2D Parallel 
p = StencilPara{Int64, 2}(L=4, IG=2)

# 4D Parallel 
p = StencilPara{Int64, 4}(L=4)

Input data

For stencil computation input can be:

  • Array:
    • e.g. 2D array of size 360000
    #generating random array data
    random_array_data = [rand(ComplexF64, (3,3)) for i in 1:360000]
  • HDF5 file:
    • File containing array with variable name "data"
    • e.g. 2D array of size 600x600x2x3x3 
    #generating random HDF5 file data
    using HDF5
    fn = "2d_l_600_cf64.h5"
    @unpack NL, master_rank, C1, C2, N, LD = p

    v = rand(ComplexF64,(600,600,2,3,3));       
    h5open(fn, "w") do f
        write(f, "data", v) 
    end

2D Sequential stencil computation with array input

    using ParallelQCDStencil
    using Parameters
    using LinearAlgebra

    p = StencilPara{Int64, 2}(L=10, master_rank=0)
    @unpack NL, master_rank, C1, C2, N, LD = p
    
    rmat = rand(ComplexF64, (LD,LD))
    
    data = ParallelQCDStencil.linear_link_values_to_site_based_array(p, [rmat for i in 1:NL])
      
    total = compute_stencil(p, data, Sequential())
    
    println("total:$(total) serial\n ")

2D Parallel stencil computation with array input

    using ParallelQCDStencil
    using MPI
    using Parameters
    using LinearAlgebra
    
    p = StencilPara{Int64, 2}(L=10, IG=5, master_rank=1)
    @unpack NL, master_rank, C1, C2, N = p
    
    MPI.Init()
    comm = MPI.COMM_WORLD
    rank = MPI.Comm_rank(comm)
    
    if (rank == master_rank)
        data = ParallelQCDStencil.linear_link_values_to_site_based_array(p, [rand(ComplexF64, (3,3)) for i in 1:NL])
    else
        data = nothing
    end
    total = compute_stencil(p, data, comm, rank, LocalGridNaive())
    
    if (rank == master_rank)
        println("total:$(total)\n")
    end
    MPI.Finalize()

2D Parallel stencil computation with HDF5 input

    using ParallelQCDStencil
    using MPI
    using Parameters
    using LinearAlgebra
    using HDF5
    
    p = StencilPara{Int64, 2}(L=10, IG=5, master_rank=1)
    @unpack NL, master_rank, C1, C2, N, L, IG = p
    d = ParallelQCDStencil._get_d_val(p)
    
    MPI.Init()
    comm = MPI.COMM_WORLD
    rank = MPI.Comm_rank(comm)
    
    file = "data/2d/test_files/2d_array_l_10_cf64.h5"
    total = compute_stencil(p, file, comm, rank, LocalGridNaive(); file=true)

    if (rank == master_rank)
        println("total:$(total)\n")
    end
    MPI.Finalize()
